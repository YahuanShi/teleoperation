# =============================================================================
# Teleoperation Requirements
# Ubuntu 22.04 + ROS 2 Humble
# UR5 + Weiss CRG 30-050 + Uarm master arm + RealSense cameras
# =============================================================================
#
# 1. Install ROS 2 Humble (managed by apt — do NOT pip-install these):
#
#   sudo apt install ros-humble-desktop \
#                    python3-colcon-common-extensions \
#                    ros-humble-cv-bridge \
#                    ros-humble-sensor-msgs \
#                    ros-humble-std-msgs
#
# 2. Build the uarm package (from teleoperation/ root):
#
#   cd teleoperation
#   colcon build --packages-select uarm
#   source install/setup.bash
#
# 3. Install pip dependencies:
#
#   pip install -r requirements.txt
# =============================================================================
